Sailors	(sid:integer, sname:string, rating:integer, age:real)
Boats (bid:integer, bname:string, color:string)
Reserves (sid:integer, bid:integer, day:date) 

SELECT 	Sailors.sname, Sailors.age 
FROM 	Sailors
WHERE 	Sailors.sid IN ( 	SELECT	sid
							FROM	Reserves
							WHERE	day = day ) 

SLECT 	sname
FROM	Sailors
WHERE	sid = sid

SELECT	Sailors.sname, Boats.bname, Boats.color
FROM	Sailors, Boats, Reserves
WHERE	Sailors.sid = Reserves.sid AND Boats.bid = Reserves.bid
